Подключите Google Sheets — один раз, 10 минут

1. Создайте новую Google таблицу — она будет хранить ответы
2. В таблице: Расширения → Apps Script — вставьте этот код и нажмите «Развернуть → Новое развёртывание → Веб-приложение»
3. Доступ: «Все» → скопируйте URL развёртывания → вставьте ниже
4. Нажмите «Сохранить» — готово! Скачайте бриф и отправьте заказчику
function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var row = [new Date(), data.clientName || '—']; data.fields.forEach(function(f) { row.push(f.label + ': ' + f.value); }); sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
